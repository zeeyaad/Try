import React, { useState, useRef } from "react";
import "./PrintStuff.css";
import { FontAwesomeIcon } from "@fortawesome/react-fontawesome";
import { faPrint, faUpload } from "@fortawesome/free-solid-svg-icons";

const PrintStuff = () => {
  const [formData, setFormData] = useState({
    fullName: "",
    jobTitle: "",
    image: null,
    imagePreview: "",
  });
  const iframeRef = useRef();
  const fileInputRef = useRef();

  const handleInputChange = (e) => {
    const { name, value } = e.target;
    setFormData((prev) => ({
      ...prev,
      [name]: value,
    }));
  };

  const handleImageUpload = (e) => {
    const file = e.target.files[0];
    if (file) {
      const reader = new FileReader();
      reader.onload = (e) => {
        setFormData((prev) => ({
          ...prev,
          image: file,
          imagePreview: e.target.result,
        }));
      };
      reader.readAsDataURL(file);
    }
  };

  const handlePrint = () => {
    if (!formData.fullName || !formData.jobTitle || !formData.imagePreview) {
      alert("Please fill in all fields and upload an image");
      return;
    }

    const iframe = iframeRef.current;
    const doc = iframe.contentWindow.document;

    const cardHtml = getCardHTML(
      formData.fullName,
      formData.imagePreview,
      formData.jobTitle,
    );

    doc.open();
    doc.write(cardHtml);
    doc.close();

    iframe.onload = () => {
      iframe.contentWindow.focus();
      iframe.contentWindow.print();
    };
  };

  return (
    <div className="print-stuff-container">
      <div className="print-stuff-card">
        <h2 className="page-title">طباعة بطاقة عضوية</h2>

        <div className="form-section">
          <div className="image-upload-section">
            <div className="image-preview">
              {formData.imagePreview ? (
                <img src={formData.imagePreview} alt="Preview" />
              ) : (
                <div className="placeholder">
                  <FontAwesomeIcon icon={faUpload} size="2x" />
                  <p>اختر صورة</p>
                </div>
              )}
            </div>
            <input
              type="file"
              ref={fileInputRef}
              onChange={handleImageUpload}
              accept="image/*"
              style={{ display: "none" }}
            />
            <button
              type="button"
              className="btn upload-btn"
              onClick={() => fileInputRef.current.click()}
            >
              <FontAwesomeIcon icon={faUpload} />
              <span className="mx-2">رفع صورة</span>
            </button>
          </div>

          <div className="form-inputs">
            <div className="form-group">
              <label htmlFor="fullName">الاسم الكامل</label>
              <input
                type="text"
                id="fullName"
                name="fullName"
                value={formData.fullName}
                onChange={handleInputChange}
                placeholder="أدخل الاسم الكامل"
                dir="rtl"
              />
            </div>

            <div className="form-group">
              <label htmlFor="jobTitle">المسمى الوظيفي</label>
              <input
                type="text"
                id="jobTitle"
                name="jobTitle"
                value={formData.jobTitle}
                onChange={handleInputChange}
                placeholder="أدخل المسمى الوظيفي"
                dir="rtl"
              />
            </div>
          </div>
        </div>

        <div className="print-section">
          <div className="print-instructions">
            <h4>تعليمات الطباعة:</h4>
            <ol dir="rtl">
              <li>
                options {">"} printer {"("}Destination{")"} {">"} MagicCard Rio
                Pro 360
              </li>
              <li>
                additional options {">"} margin {">"} none
              </li>
              <li>
                options {">"} orientation {">"} portrait
              </li>
              <li>
                options {">"} paper size {">"} cr80
              </li>
            </ol>
          </div>

          <button
            onClick={handlePrint}
            className="btn print-btn"
            disabled={
              !formData.fullName || !formData.jobTitle || !formData.imagePreview
            }
          >
            <FontAwesomeIcon icon={faPrint} />
            <span className="mx-2">طباعة البطاقة</span>
          </button>
        </div>

        <div style={{ display: "none" }}>
          <iframe ref={iframeRef} id="printFrame" title="Print Frame" />
        </div>
      </div>
    </div>
  );
};

// Template function (same as CardPrint component)

          </div>

          </div>
        </div>

        <!-- signature inside right panel -->
        <div class="executive-director-signature" dir="rtl">
          <div class="executive-director-title">المدير التنفيذي</div>
          <div class="executive-director-name">ا.د / احمد فاروق</div>

      </section>
    </div>
  </body>

};

export default PrintStuff;
